        // === MENU SERVICES ===
        JMenu mServices = creerMenuSimple("Services");
        
        JMenuItem iGestionRestaurants = new JMenuItem("Gestion des Restaurants");
        iGestionRestaurants.addActionListener(e -> {
            if (gp.getServiceManager() != null) {
                new DialogGestionRestaurants(MainFrame.this, gp.getServiceManager(), gp).setVisible(true);
                rafraichirTousLesPanels();
            } else {
                JOptionPane.showMessageDialog(MainFrame.this, "ServiceManager non initialisé", "Erreur", JOptionPane.ERROR_MESSAGE);
            }
        });
        
        JMenuItem iGestionBoutiques = new JMenuItem("Gestion des Boutiques");
        iGestionBoutiques.addActionListener(e -> {
            if (gp.getServiceManager() != null) {
                new DialogGestionBoutiques(MainFrame.this, gp.getServiceManager(), gp).setVisible(true);
                rafraichirTousLesPanels();
            } else {
                JOptionPane.showMessageDialog(MainFrame.this, "ServiceManager non initialisé", "Erreur", JOptionPane.ERROR_MESSAGE);
            }
        });
        
        mServices.add(iGestionRestaurants);
        mServices.addSeparator();
        mServices.add(iGestionBoutiques);
        mb.add(mServices);
